void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    // Normalized pixel coordinates (from 0 to 1)
    vec2 uv = fragCoord/iResolution.xy;
   
    float scaleValue = 1.5;
    vec2 scaleVector = uv * scaleValue;
    
    float rotationAngle = -0.785398;
    vec2 rotationVector = vec2(scaleVector.x * cos(rotationAngle) - scaleVector.y * sin(rotationAngle),
                          scaleVector.x * sin(rotationAngle) + scaleVector.y * cos(rotationAngle));
    
    vec2 translationVector = rotationVector + vec2(1.0,1.0);
     
    // Time varying pixel color
    vec4 t = texture(iChannel0,translationVector);

    // Output to screen
    fragColor = t;
}
